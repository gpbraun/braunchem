\textbf{Assinale} a alternativa que mais se aproxima do número de átomos de ferro no imã???

\begin{choices}

    \item \(\pu{2,7E23}\)
    \item \(\pu{1,6E24}\)
    \item \(\pu{9,4E24}\)
    \item \(\pu{5,5E25}\)
\end{choices}

\end{problem}
